    seajs加载jquery时提示$ is not a function该怎么解决

    
 if (typeof define === "function" && (define.amd)) {
   define( "jquery", [], function() {
     return jQuery;
   });
 }
改成

 if (typeof define === "function" && (define.amd || define.cmd)) {
   define( "jquery", [], function() {
     return jQuery;
   });
 }
或

 if (typeof define === "function") {
   define( "jquery", [], function() {
     return jQuery;
   });
 }
